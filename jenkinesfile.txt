pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build') {
            steps {
                sh 'docker build -t my-app .'
            }
        }

        stage('Test') {
            steps {
                sh 'docker run my-app npm test'
            }
        }

        stage('Deploy') {
            steps {
                sh 'docker push my-app:latest'
            }
        }
    }
}
